cmake_minimum_required(VERSION 3.10)#支持c++11的版本
project(ThreadPool LANGUAGES CXX)#c++实现

#设置默认构造类型为 Debug
if(NOT CMAKE_BUILD_TYPE)
  set(CMAKE_BUILD_TYPE Debug)
endif()

#设置输出路径为当前目录的 bin文件夹,没有就会自动创建
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${PROJECT_BINARY_DIR}/bin)

#设置编译器标志
set(CMAKE_CXX_STANDARD 11)#c++11标准
set(CMAKE_CXX_STANDARD_REQUIRED ON)#要求使用c++11标准
set(CMAKE_CXX_EXTENSIONS OFF)#禁用扩展
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wall")#开启所有警告
set(CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} -O0")
set(CMAKE_CXX_FLAGS_RELEASE "${CMAKE_CXX_FLAGS_RELEASE} -O2 -DNDEBUG")

#打印头文件标志
message(STATUS "CXX_FLAGS: $")


add_executable(ThreadPool_test main.cc  ThreadPool.h)


target_link_libraries(ThreadPool pthread)
